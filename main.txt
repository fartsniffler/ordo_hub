local function createESP(humanoid, character)
	local billboard = Instance.new("BillboardGui")
	billboard.Adornee = character:WaitForChild("Head")
	billboard.Size = UDim2.new(0, 100, 0, 50) -- Keep this size, the actual display will be scaled by TextSize
	billboard.StudsOffset = Vector3.new(0, 5, 0)
	billboard.AlwaysOnTop = true
	billboard.MaxDistance = math.huge

	local textLabel = Instance.new("TextLabel", billboard)
	textLabel.Size = UDim2.new(1, 0, 1, 0)
	textLabel.BackgroundTransparency = 1
	textLabel.TextColor3 = Color3.new(1, 1, 1)
	textLabel.TextStrokeTransparency = 0.5
	textLabel.TextWrapped = false
	textLabel.Font = Enum.Font.SourceSansBold
	textLabel.TextScaled = false -- Disable automatic scaling
	textLabel.TextSize = espSize -- Manually increase text size (default is usually around 14)

	billboard.Parent = character

	-- Update ESP info
	local function updateESP()
		if humanoid and humanoid.Parent and entityESP == true and character:FindFirstChild("HumanoidRootPart") then
			local distance = (camera.CFrame.Position - character.HumanoidRootPart.Position).Magnitude
			textLabel.Text = string.format("%s | %d/%d | %.1f", character.Name, humanoid.Health, humanoid.MaxHealth, distance)
			textLabel.TextSize = espSize
		else
			billboard:Destroy()
		end
	end

	-- Connect to Humanoid's health change
	humanoid.HealthChanged:Connect(updateESP)
	game:GetService("RunService").RenderStepped:Connect(updateESP)
end
